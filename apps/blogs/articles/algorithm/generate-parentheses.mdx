[22. 括号生成](https://leetcode.cn/problems/generate-parentheses/)

数字 n 代表生成括号的对数，请你设计一个函数，用于能够生成所有可能的并且 有效的 括号组合。

```typescript
function generateParenthesis(n: number): string[] {
  const answer: string[] = [];
  if (n < 1) {
    return answer;
  }
  dfs("", n, n, answer);
  return answer;
}

function dfs(str: string, left: number, right: number, array: string[]) {
  if (left === 0 && right === 0) {
    array.push(str);
    return;
  }

  if (left > right) return;

  left > 0 && dfs(str + "(", left - 1, right, array);
  right > 0 && dfs(str + ")", left, right - 1, array);
}
```
